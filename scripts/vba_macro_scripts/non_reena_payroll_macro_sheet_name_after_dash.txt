'========================
' Main entry
'========================
Sub ProcessAllNonReenaPayrolls_SheetNameAfterDash()
    Const EXCLUDED_PATTERN As String = "REENA"
    Const COL_DATE As Long = 1
    Const COL_START As Long = 4
    Const COL_END As Long = 5
    Const COL_RATE As Long = 6
    Const COL_TOTAL As Long = 7
    Const COL_DURATION As Long = 8

    Dim wsSource As Worksheet
    Dim lastRow As Long, colClient As Long
    Dim rowIdx As Long
    Dim clientIndex As Long
    Dim rawClient As String
    Dim clientMap As New Collection
    Dim sortedClients As Variant

    Set wsSource = ActiveSheet
    colClient = 3
    lastRow = wsSource.Cells(wsSource.Rows.Count, colClient).End(xlUp).Row

    PrepareSourceSheetForInvoiceProcessing wsSource, 2, lastRow, COL_DATE, COL_START, COL_END, COL_RATE, COL_TOTAL, COL_DURATION

    For rowIdx = 2 To lastRow
        rawClient = NormalizeClientName(GetSafeTextValue(wsSource.Cells(rowIdx, colClient).Value))
        If Len(rawClient) > 0 Then
            If InStr(1, rawClient, EXCLUDED_PATTERN, vbTextCompare) = 0 Then
                If Not CollectionContains(clientMap, rawClient) Then
                    clientMap.Add rawClient
                End If
            End If
        End If
    Next rowIdx

    If clientMap.Count = 0 Then
        MsgBox "No non-""" & EXCLUDED_PATTERN & """ clients were found.", vbInformation
        Exit Sub
    End If

    sortedClients = GetSortedCollectionValues(clientMap)
    nextInvoiceNumber = STARTING_INVOICE_NUMBER

    For clientIndex = LBound(sortedClients) To UBound(sortedClients)
        ProcessSingleClientNonReena wsSource, CStr(sortedClients(clientIndex))
    Next clientIndex

    ReorderSheetsAlphabetically sortedClients

    MsgBox clientMap.Count & " non-""" & EXCLUDED_PATTERN & """ client(s) processed.", vbInformation
End Sub

'========================
' Per-client processing
'========================
Private Sub ProcessSingleClientNonReena(ByVal wsSource As Worksheet, ByVal targetClient As String)
    Const TABLE_HEADER_ROW As Long = 13
    Const HST_RATE As Double = 0.13

    Dim wsDest As Worksheet
    Dim invoiceNumber As Long
    Dim lastRow As Long, lastCol As Long, destRow As Long
    Dim colDate As Long, colName As Long, colClient As Long
    Dim colStart As Long, colEnd As Long, colRate As Long
    Dim colTotal As Long, colDuration As Long, colUnit As Long, colONS As Long
    Dim colHelper As Long, colSortDate As Long, colSortTime As Long
    Dim colSortHours As Long, colShiftSequence As Long
    Dim i As Long

    lastRow = wsSource.Cells(wsSource.Rows.Count, 1).End(xlUp).Row
    lastCol = wsSource.Cells(1, wsSource.Columns.Count).End(xlToLeft).Column

    colDate = 1
    colName = 2
    colClient = 3
    colStart = 4
    colEnd = 5
    colRate = 6
    colTotal = 7
    colDuration = 8
    colUnit = 9
    colONS = 10

    Dim sheetName As String
    sheetName = BuildClientSheetName(targetClient)

    Dim companyName As String
    Dim itemDescription As String
    companyName = ResolveCompanyName(targetClient)
    itemDescription = ResolveItemDescription(targetClient)
    invoiceNumber = GetNextInvoiceNumber()

    On Error Resume Next
    Set wsDest = ThisWorkbook.Sheets(sheetName)
    On Error GoTo 0

    If wsDest Is Nothing Then
        Set wsDest = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsDest.Name = sheetName
    Else
        wsDest.Cells.Clear
    End If

    wsSource.Rows(1).Copy wsDest.Rows(1)

    destRow = TABLE_HEADER_ROW + 1
    wsDest.Columns(colDate).NumberFormat = "dd/mm/yyyy"

    For i = 2 To lastRow
        If NormalizeClientName(GetSafeTextValue(wsSource.Cells(i, colClient).Value)) = targetClient Then

            Dim isONS As Boolean
            isONS = (UCase$(Trim$(GetSafeTextValue(wsSource.Cells(i, colONS).Value))) = "YES")

            If isONS Then
                Dim originalEnd As Variant
                Dim originalShiftDate As Variant
                originalEnd = NormalizeTimeValue(wsSource.Cells(i, colEnd).Value)
                originalShiftDate = NormalizeDate(wsSource.Cells(i, colDate).Value)

                wsDest.Rows(destRow).Value = wsSource.Rows(i).Value
                wsDest.Cells(destRow, colDate).Value = originalShiftDate
                wsDest.Cells(destRow, colStart).Value = TimeValue("11:00:00 PM")
                wsDest.Cells(destRow, colEnd).Value = TimeValue("6:00:00 AM")
                wsDest.Cells(destRow, colDuration).Value = 1
                wsDest.Cells(destRow, colClient).Value = itemDescription
                wsDest.Cells(destRow, colUnit).Value = CStr(wsDest.Cells(destRow, colUnit).Value) & " - ONS"
                wsDest.Cells(destRow, colTotal).Formula = "=ROUND(" & wsDest.Cells(destRow, colDuration).Address(False, False) & "*" & wsDest.Cells(destRow, colRate).Address(False, False) & ",2)"
                destRow = destRow + 1

                wsDest.Rows(destRow).Value = wsSource.Rows(i).Value
                wsDest.Cells(destRow, colDate).Value = originalShiftDate
                wsDest.Cells(destRow, colStart).Value = TimeValue("6:00:00 AM")
                wsDest.Cells(destRow, colEnd).Value = originalEnd

                Dim rate2 As Double
                If InStr(1, UCase$(GetSafeTextValue(wsSource.Cells(i, colName).Value)), "MEDS", vbTextCompare) > 0 Then
                    rate2 = 27.22
                Else
                    rate2 = 26.88
                End If
                wsDest.Cells(destRow, colRate).Value = rate2

                Dim dur As Double
                If IsDate(originalEnd) And originalEnd < TimeValue("6:00:00 AM") Then
                    dur = (originalEnd + 1) - TimeValue("6:00:00 AM")
                Else
                    dur = originalEnd - TimeValue("6:00:00 AM")
                End If
                wsDest.Cells(destRow, colDuration).Value = Round(dur * 24, 2)

                wsDest.Cells(destRow, colClient).Value = itemDescription
                wsDest.Cells(destRow, colUnit).Value = ""
                wsDest.Cells(destRow, colTotal).Formula = "=ROUND(" & wsDest.Cells(destRow, colDuration).Address(False, False) & "*" & wsDest.Cells(destRow, colRate).Address(False, False) & ",2)"
                destRow = destRow + 1

            Else
                Dim normalizedShiftDate As Variant
                normalizedShiftDate = NormalizeDate(wsSource.Cells(i, colDate).Value)

                wsDest.Rows(destRow).Value = wsSource.Rows(i).Value
                wsDest.Cells(destRow, colDate).Value = normalizedShiftDate

                Dim startTime As Variant, endTime As Variant, duration As Double
                startTime = NormalizeTimeValue(wsSource.Cells(i, colStart).Value)
                endTime = NormalizeTimeValue(wsSource.Cells(i, colEnd).Value)
                wsDest.Cells(destRow, colStart).Value = startTime
                wsDest.Cells(destRow, colEnd).Value = endTime

                If endTime < startTime Then
                    duration = (endTime + 1) - startTime
                Else
                    duration = endTime - startTime
                End If
                wsDest.Cells(destRow, colDuration).Value = Round(duration * 24, 2)

                wsDest.Cells(destRow, colClient).Value = itemDescription
                wsDest.Cells(destRow, colTotal).Formula = "=ROUND(" & wsDest.Cells(destRow, colDuration).Address(False, False) & "*" & wsDest.Cells(destRow, colRate).Address(False, False) & ",2)"
                destRow = destRow + 1
            End If
        End If
    Next i

    If destRow = TABLE_HEADER_ROW + 1 Then
        wsDest.Cells.Clear
        wsDest.Name = BuildSheetName(sheetName & " (Empty)")
        Exit Sub
    End If

    For i = TABLE_HEADER_ROW + 1 To destRow - 1
        wsDest.Cells(i, colName).Value = Replace(CStr(wsDest.Cells(i, colName).Value), "@", "")
    Next i

    colHelper = lastCol + 1
    colSortDate = lastCol + 2
    colSortTime = lastCol + 3
    colSortHours = lastCol + 4
    colShiftSequence = lastCol + 5

    For i = TABLE_HEADER_ROW + 1 To destRow - 1
        Dim unitValue As String
        Dim isFollowingONSRow As Boolean
        unitValue = CStr(wsDest.Cells(i, colUnit).Value)
        isFollowingONSRow = False

        If unitValue = "" And i > TABLE_HEADER_ROW + 1 Then
            isFollowingONSRow = (InStr(1, CStr(wsDest.Cells(i - 1, colUnit).Value), "ONS", vbTextCompare) > 0)
        End If

        If unitValue <> "" Then
            wsDest.Cells(i, colHelper).Value = Replace(unitValue, " - ONS", "")
        ElseIf i > TABLE_HEADER_ROW + 1 Then
            wsDest.Cells(i, colHelper).Value = wsDest.Cells(i - 1, colHelper).Value
        End If

        If isFollowingONSRow Then
            wsDest.Cells(i, colSortDate).Value = wsDest.Cells(i - 1, colSortDate).Value
            wsDest.Cells(i, colSortTime).Value = wsDest.Cells(i - 1, colSortTime).Value
            wsDest.Cells(i, colSortHours).Value = GetSortHoursValue(wsDest.Cells(i - 1, colDuration).Value) + _
                                                 GetSortHoursValue(wsDest.Cells(i, colDuration).Value)
            wsDest.Cells(i, colShiftSequence).Value = 2
            wsDest.Cells(i - 1, colSortHours).Value = wsDest.Cells(i, colSortHours).Value
        Else
            wsDest.Cells(i, colSortDate).Value = GetSortDateValue(wsDest.Cells(i, colDate).Value)
            wsDest.Cells(i, colSortTime).Value = GetSortTimeValue(wsDest.Cells(i, colStart).Value)
            wsDest.Cells(i, colSortHours).Value = GetSortHoursValue(wsDest.Cells(i, colDuration).Value)
            wsDest.Cells(i, colShiftSequence).Value = 1
        End If
    Next i

    ApplyPostBuildOrdering wsDest, TABLE_HEADER_ROW + 1, destRow - 1, colHelper, colSortDate, colSortTime, colSortHours, colName, colShiftSequence

    wsDest.Rows(TABLE_HEADER_ROW).Value = wsDest.Rows(1).Value
    wsDest.Rows(1).Clear

    wsDest.Cells(TABLE_HEADER_ROW, colName).Value = "Recorded by Name"
    wsDest.Cells(TABLE_HEADER_ROW, colClient).Value = "Item Description"
    wsDest.Cells(TABLE_HEADER_ROW, colRate).Value = "Rate"
    wsDest.Cells(TABLE_HEADER_ROW, colTotal).Value = "Amount"
    wsDest.Cells(TABLE_HEADER_ROW, colUnit).Value = "Code"

    With wsDest.Range(wsDest.Cells(TABLE_HEADER_ROW, 1), wsDest.Cells(TABLE_HEADER_ROW, lastCol))
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .Borders.LineStyle = xlContinuous
        .Borders.Weight = xlMedium
    End With

    wsDest.Range(wsDest.Cells(TABLE_HEADER_ROW + 1, colDate), wsDest.Cells(destRow - 1, colDate)).NumberFormat = "dd/mm/yyyy"
    wsDest.Range(wsDest.Cells(TABLE_HEADER_ROW + 1, colStart), wsDest.Cells(destRow - 1, colEnd)).NumberFormat = "h:mm:ss AM/PM"

    wsDest.Columns(colRate).NumberFormat = "0.00"
    wsDest.Columns(colTotal).NumberFormat = "#,##0.00"

    With wsDest.Range(wsDest.Cells(TABLE_HEADER_ROW, 1), wsDest.Cells(destRow - 1, lastCol))
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .Font.Name = "Arial"
        .Font.Size = 14
        .Font.Bold = False
    End With
    wsDest.Rows(TABLE_HEADER_ROW).Font.Bold = True

    ApplyInvoiceHeaderNonReena wsDest, TABLE_HEADER_ROW, companyName, itemDescription, invoiceNumber

    Dim currentUnit As String
    Dim r As Long, startRow As Long

    r = TABLE_HEADER_ROW + 1
    currentUnit = CStr(wsDest.Cells(r, colHelper).Value)
    startRow = r

    Do While r < destRow
        If CStr(wsDest.Cells(r, colHelper).Value) = currentUnit Then
            r = r + 1
        Else
            ApplyUnitSectionFormatting wsDest, startRow, r - 1, lastCol
            InsertSubtotalRow wsDest, r, startRow, colName, colDuration, colTotal, lastCol, currentUnit
            currentUnit = CStr(wsDest.Cells(r + 1, colHelper).Value)
            startRow = r + 1
            destRow = destRow + 1
            r = r + 1
        End If
    Loop

    InsertSubtotalRow wsDest, r, startRow, colName, colDuration, colTotal, lastCol, currentUnit
    ApplyUnitSectionFormatting wsDest, startRow, r - 1, lastCol

    Dim totalsRow As Long
    totalsRow = destRow + 1

    AddFinalTotalsBox_SumOnlySubtotalRows wsDest, TABLE_HEADER_ROW + 1, destRow, colName, colDuration, colTotal, totalsRow, HST_RATE
    AddFooterCodesAndPaymentInfo wsDest, totalsRow

    wsDest.Rows.RowHeight = 22
    wsDest.Rows(7).RowHeight = 34.5

    wsDest.Columns(colShiftSequence).Delete
    wsDest.Columns(colSortHours).Delete
    wsDest.Columns(colSortTime).Delete
    wsDest.Columns(colSortDate).Delete
    wsDest.Columns(colHelper).Delete
    ApplyFixedColumnWidths wsDest

    SetPrintAreaThroughEnquiries wsDest
    ApplyPrintSetup wsDest
    InsertCompanyLogo wsDest
    AddInvoiceExportButtonsForSheet wsDest
End Sub

'========================
' Print setup
'========================
Private Sub ApplyPrintSetup(ByVal ws As Worksheet)
    Dim lastPrintRow As Long
    lastPrintRow = GetEnquiriesRow(ws)
    If lastPrintRow = 0 Then lastPrintRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

    With ws.PageSetup
        .Orientation = xlLandscape
        .TopMargin = Application.InchesToPoints(0.75)
        .BottomMargin = Application.InchesToPoints(0.75)
        .LeftMargin = Application.InchesToPoints(0.25)
        .RightMargin = Application.InchesToPoints(0.25)
        .HeaderMargin = Application.InchesToPoints(0.3)
        .FooterMargin = Application.InchesToPoints(0.3)
        .Zoom = False
        .FitToPagesWide = 1
        .FitToPagesTall = False
    End With

    ws.ResetAllPageBreaks
    AddHorizontalPageBreaksEveryNRows ws, 41, 1, lastPrintRow
End Sub

Private Sub AddHorizontalPageBreaksEveryNRows(ByVal ws As Worksheet, ByVal rowsPerPage As Long, ByVal startRow As Long, ByVal endRow As Long)
    Dim rr As Long
    If rowsPerPage <= 0 Then Exit Sub
    If endRow <= startRow Then Exit Sub

    rr = startRow + rowsPerPage
    Do While rr <= endRow
        ws.HPageBreaks.Add Before:=ws.Rows(rr)
        rr = rr + rowsPerPage
    Loop
End Sub

Private Sub AddFinalTotalsBox_SumOnlySubtotalRows( _
    ByVal ws As Worksheet, _
    ByVal firstRow As Long, _
    ByVal lastRow As Long, _
    ByVal colName As Long, _
    ByVal colDuration As Long, _
    ByVal colAmount As Long, _
    ByVal startRow As Long, _
    ByVal hstRate As Double)

    Const COL_LABEL As Long = 6
    Const COL_AMT As Long = 7
    Const COL_DUR As Long = 8

    Dim criteriaRange As Range, amtRange As Range, durRange As Range
    Set criteriaRange = ws.Range(ws.Cells(firstRow, colName), ws.Cells(lastRow, colName))
    Set amtRange = ws.Range(ws.Cells(firstRow, colAmount), ws.Cells(lastRow, colAmount))
    Set durRange = ws.Range(ws.Cells(firstRow, colDuration), ws.Cells(lastRow, colDuration))

    ws.Cells(startRow, COL_LABEL).Value = "SUBTOTAL"
    ws.Cells(startRow, COL_AMT).Formula = "=ROUND(SUMIF(" & criteriaRange.Address(False, False) & ",""Subtotal for Unit*""," & amtRange.Address(False, False) & "),2)"
    ws.Cells(startRow, COL_DUR).Formula = "=ROUND(SUMIF(" & criteriaRange.Address(False, False) & ",""Subtotal for Unit*""," & durRange.Address(False, False) & "),2)"

    ws.Cells(startRow + 1, COL_LABEL).Value = "HST 13%"
    ws.Cells(startRow + 1, COL_AMT).Formula = "=ROUND(" & ws.Cells(startRow, COL_AMT).Address(False, False) & "*" & CStr(hstRate) & ",2)"
    ws.Cells(startRow + 1, COL_DUR).Value = ""

    ws.Cells(startRow + 2, COL_LABEL).Value = "TOTAL"
    ws.Cells(startRow + 2, COL_AMT).Formula = "=" & ws.Cells(startRow, COL_AMT).Address(False, False) & "+" & ws.Cells(startRow + 1, COL_AMT).Address(False, False)
    ws.Cells(startRow + 2, COL_DUR).Value = ""

    Dim rngBox As Range
    Set rngBox = ws.Range(ws.Cells(startRow, COL_LABEL), ws.Cells(startRow + 2, COL_DUR))

    With rngBox
        .Font.Name = "Arial"
        .Font.Size = 14
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
    End With

    ws.Range(ws.Cells(startRow, COL_AMT), ws.Cells(startRow + 2, COL_AMT)).NumberFormat = "#,##0.00"

    With rngBox.Borders
        .LineStyle = xlContinuous
        .Weight = xlMedium
    End With

    ClearTotalsDurationCellBorders ws, startRow + 1, COL_AMT, COL_DUR
    ClearTotalsDurationCellBorders ws, startRow + 2, COL_AMT, COL_DUR
End Sub

Private Sub ClearTotalsDurationCellBorders(ByVal ws As Worksheet, ByVal rowNumber As Long, ByVal amountCol As Long, ByVal durationCol As Long)

    With ws.Cells(rowNumber, amountCol).Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .Weight = xlMedium
    End With

    With ws.Cells(rowNumber, amountCol).Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .Weight = xlMedium
    End With

    With ws.Cells(rowNumber, amountCol).Borders(xlEdgeBottom)
        .LineStyle = xlContinuous
        .Weight = xlMedium
    End With
End Sub

'========================
' Print area A:I through ENQUIRIES row
'========================
Private Sub SetPrintAreaThroughEnquiries(ByVal ws As Worksheet)
    Dim lastRow As Long
    lastRow = GetEnquiriesRow(ws)
    If lastRow = 0 Then lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    ws.PageSetup.PrintArea = ws.Range("A1:I" & lastRow).Address
End Sub

Private Function GetEnquiriesRow(ByVal ws As Worksheet) As Long
    Dim f As Range
    On Error Resume Next
    Set f = ws.Columns("A").Find(What:="ENQUIRIES:", LookIn:=xlValues, LookAt:=xlWhole, _
                                 SearchOrder:=xlByRows, SearchDirection:=xlPrevious, MatchCase:=False)
    On Error GoTo 0

    If f Is Nothing Then
        GetEnquiriesRow = 0
    Else
        GetEnquiriesRow = f.Row
    End If
End Function

'========================
' Borders + subtotal row
'========================
Private Sub ApplyUnitSectionFormatting(ws As Worksheet, startRow As Long, endRow As Long, lastCol As Long)
    Dim c As Long
    For c = 1 To lastCol
        With ws.Range(ws.Cells(startRow, c), ws.Cells(endRow, c)).Borders(xlEdgeLeft)
            .LineStyle = xlContinuous
            .Weight = xlMedium
        End With
        With ws.Range(ws.Cells(startRow, c), ws.Cells(endRow, c)).Borders(xlEdgeRight)
            .LineStyle = xlContinuous
            .Weight = xlMedium
        End With
    Next c
End Sub

Private Sub InsertSubtotalRow(ws As Worksheet, insertRow As Long, startRow As Long, _

    End If
End Function

    pos = InStr(fullClient, " - ")
    If pos > 0 Then
        GetCompanyName = Left$(fullClient, pos - 1)
    Else
        GetCompanyName = fullClient
    End If
End Function

Private Function BuildClientSheetName(ByVal fullClient As String) As String
    Dim clientSuffix As String

    clientSuffix = Trim$(GetDisplayClientName(fullClient))
    If clientSuffix = "" Then clientSuffix = "Unnamed Client"

    BuildClientSheetName = BuildSheetName(clientSuffix)
End Function

'========================
' Sheet name sanitizer
'========================
Private Function BuildSheetName(ByVal fullClient As String) As String
    Dim sanitized As String
    sanitized = Replace(fullClient, ":", "-")
    sanitized = Replace(sanitized, "/", "-")
    sanitized = Replace(sanitized, "\", "-")
    sanitized = Replace(sanitized, "?", "-")
    If Len(sanitized) > 31 Then sanitized = Left$(sanitized, 31)
    BuildSheetName = sanitized
End Function

Private Function NormalizeClientName(ByVal rawClient As String) As String
    Dim normalized As String

    normalized = Trim$(rawClient)
    normalized = Replace(normalized, " - ", "|")
    normalized = Replace(normalized, " -", "|")
    normalized = Replace(normalized, "- ", "|")
    normalized = Replace(normalized, "-", "|")
    normalized = Replace(normalized, "|", " - ")

    Do While InStr(normalized, "  ") > 0
        normalized = Replace(normalized, "  ", " ")
    Loop

    NormalizeClientName = Trim$(normalized)
End Function

    If IsError(val) Then Exit Function
    If IsNull(val) Then Exit Function
    If IsEmpty(val) Then Exit Function

    GetSafeTextValue = Trim$(CStr(val))

Private Sub PrepareSourceSheetForInvoiceProcessing(ByVal ws As Worksheet, ByVal firstDataRow As Long, ByVal lastDataRow As Long, _
    ByVal colDate As Long, ByVal colStart As Long, ByVal colEnd As Long, ByVal colRate As Long, _
    ByVal colTotal As Long, ByVal colDuration As Long)

    Dim rowNumber As Long
    Dim parsedDate As Date
    Dim parsedTime As Double
    Dim parsedNumber As Double
    Dim rawDateText As String

    If lastDataRow < firstDataRow Then Exit Sub

    For rowNumber = firstDataRow To lastDataRow
        rawDateText = Trim$(ws.Cells(rowNumber, colDate).Text)

        If rawDateText <> "" Then
            If TryParseExcelDate(rawDateText, parsedDate) Then
                ws.Cells(rowNumber, colDate).Value = parsedDate
            End If
        ElseIf TryParseExcelDate(ws.Cells(rowNumber, colDate).Value, parsedDate) Then
            ws.Cells(rowNumber, colDate).Value = parsedDate
        End If

        If TryParseExcelTime(ws.Cells(rowNumber, colStart).Value, parsedTime) Then
            ws.Cells(rowNumber, colStart).Value = parsedTime
        End If

        If TryParseExcelTime(ws.Cells(rowNumber, colEnd).Value, parsedTime) Then
            ws.Cells(rowNumber, colEnd).Value = parsedTime
        End If

        If TryParseDecimalValue(ws.Cells(rowNumber, colRate).Value, parsedNumber) Then
            ws.Cells(rowNumber, colRate).Value = parsedNumber
        End If

        If TryParseDecimalValue(ws.Cells(rowNumber, colTotal).Value, parsedNumber) Then
            ws.Cells(rowNumber, colTotal).Value = parsedNumber
        End If

        If TryParseDecimalValue(ws.Cells(rowNumber, colDuration).Value, parsedNumber) Then
            ws.Cells(rowNumber, colDuration).Value = parsedNumber
        End If
    Next rowNumber

    ws.Columns(colDate).NumberFormat = "dd/mm/yyyy"
    ws.Range(ws.Cells(firstDataRow, colStart), ws.Cells(lastDataRow, colEnd)).NumberFormat = "h:mm:ss AM/PM"
    ws.Columns(colRate).NumberFormat = "0.00"
    ws.Columns(colTotal).NumberFormat = "#,##0.00"
    ws.Columns(colDuration).NumberFormat = "0.00"
End Sub

Private Function TryParseExcelDate(ByVal val As Variant, ByRef parsedDate As Date) As Boolean
    Dim cleanedValue As String
    Dim dateParts() As String
    Dim dayPart As Long
    Dim monthPart As Long
    Dim yearPart As Long
    Dim coercedDate As Date
    Dim ddmmyyyyDate As Date
    Dim mmddyyyyDate As Date
    Dim hasDdMmCandidate As Boolean
    Dim hasMmDdCandidate As Boolean

    If IsError(val) Or IsNull(val) Or IsEmpty(val) Then Exit Function

    If IsDate(val) And (VarType(val) = vbDate Or IsNumeric(val)) Then
        coercedDate = CDate(val)
        parsedDate = DateSerial(Year(coercedDate), Month(coercedDate), Day(coercedDate))
        TryParseExcelDate = True
        Exit Function
    End If

    cleanedValue = Trim$(CStr(val))
    cleanedValue = Replace(cleanedValue, "-", "/")
    cleanedValue = Replace(cleanedValue, ".", "/")
    dateParts = Split(cleanedValue, "/")

    If UBound(dateParts) = 2 Then
        If IsNumeric(dateParts(0)) And IsNumeric(dateParts(1)) And IsNumeric(dateParts(2)) Then
            dayPart = CLng(dateParts(0))
            monthPart = CLng(dateParts(1))
            yearPart = CLng(dateParts(2))

            If yearPart < 100 Then yearPart = 2000 + yearPart

            hasDdMmCandidate = TryBuildDate(yearPart, monthPart, dayPart, ddmmyyyyDate)
            hasMmDdCandidate = TryBuildDate(yearPart, dayPart, monthPart, mmddyyyyDate)

            If hasDdMmCandidate Or hasMmDdCandidate Then
                parsedDate = ChooseBestParsedDate(ddmmyyyyDate, hasDdMmCandidate, mmddyyyyDate, hasMmDdCandidate)
                TryParseExcelDate = True
                Exit Function
            End If
        End If
    End If

    If IsDate(cleanedValue) Then
        coercedDate = CDate(cleanedValue)
        parsedDate = DateSerial(Year(coercedDate), Month(coercedDate), Day(coercedDate))
        TryParseExcelDate = True
    End If
End Function

Private Function TryBuildDate(ByVal yearPart As Long, ByVal monthPart As Long, ByVal dayPart As Long, ByRef builtDate As Date) As Boolean
    On Error Resume Next
    builtDate = DateSerial(yearPart, monthPart, dayPart)
    TryBuildDate = (Err.Number = 0)
    On Error GoTo 0
End Function

Private Function ChooseBestParsedDate(ByVal ddmmyyyyDate As Date, ByVal hasDdMmCandidate As Boolean, _
    ByVal mmddyyyyDate As Date, ByVal hasMmDdCandidate As Boolean) As Date

    Dim referenceDate As Variant

    referenceDate = GetManualInvoiceDateValue()

    If hasDdMmCandidate And Not hasMmDdCandidate Then
        ChooseBestParsedDate = ddmmyyyyDate
        Exit Function
    End If

    If hasMmDdCandidate And Not hasDdMmCandidate Then
        ChooseBestParsedDate = mmddyyyyDate
        Exit Function
    End If

    If Not IsDate(referenceDate) Then
        ChooseBestParsedDate = ddmmyyyyDate
        Exit Function
    End If

    If Abs(CDbl(ddmmyyyyDate) - CDbl(CDate(referenceDate))) <= Abs(CDbl(mmddyyyyDate) - CDbl(CDate(referenceDate))) Then
        ChooseBestParsedDate = ddmmyyyyDate
    Else
        ChooseBestParsedDate = mmddyyyyDate
    End If
End Function

Private Function TryParseExcelTime(ByVal val As Variant, ByRef parsedTime As Double) As Boolean
    Dim numericValue As Double

    If IsError(val) Or IsNull(val) Or IsEmpty(val) Then Exit Function

    If IsDate(val) Then
        parsedTime = CDbl(TimeValue(CDate(val)))
        TryParseExcelTime = True
        Exit Function
    End If

    If IsNumeric(val) Then
        numericValue = CDbl(val)
        parsedTime = numericValue - Fix(numericValue)
        TryParseExcelTime = True
        Exit Function
    End If

    If IsDate(Trim$(CStr(val))) Then
        parsedTime = CDbl(TimeValue(Trim$(CStr(val))))
        TryParseExcelTime = True
    End If
End Function

Private Function TryParseDecimalValue(ByVal val As Variant, ByRef parsedNumber As Double) As Boolean
    Dim cleanedValue As String

    If IsError(val) Or IsNull(val) Or IsEmpty(val) Then Exit Function

    If IsNumeric(val) Then
        parsedNumber = CDbl(val)
        TryParseDecimalValue = True
        Exit Function
    End If

    cleanedValue = Trim$(CStr(val))
    cleanedValue = Replace(cleanedValue, ",", "")
    cleanedValue = Replace(cleanedValue, "$", "")

    If IsNumeric(cleanedValue) Then
        parsedNumber = CDbl(cleanedValue)
        TryParseDecimalValue = True
    End If
End Function

Private Function GetSortHoursValue(ByVal shiftHours As Variant) As Double
    If IsNumeric(shiftHours) Then

    Else
        GetSortHoursValue = 0

End Function

Private Sub ApplyPostBuildOrdering(ByVal wsDest As Worksheet, ByVal firstDataRow As Long, ByVal lastDataRow As Long, _
    ByVal colHelper As Long, ByVal colSortDate As Long, ByVal colSortTime As Long, ByVal colSortHours As Long, _
    ByVal colName As Long, ByVal colShiftSequence As Long)

    If lastDataRow < firstDataRow Then Exit Sub

    With wsDest.Sort
        .SortFields.Clear
        .SortFields.Add Key:=wsDest.Range(wsDest.Cells(firstDataRow, colHelper), wsDest.Cells(lastDataRow, colHelper)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsDest.Range(wsDest.Cells(firstDataRow, colSortDate), wsDest.Cells(lastDataRow, colSortDate)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsDest.Range(wsDest.Cells(firstDataRow, colSortTime), wsDest.Cells(lastDataRow, colSortTime)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsDest.Range(wsDest.Cells(firstDataRow, colSortHours), wsDest.Cells(lastDataRow, colSortHours)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsDest.Range(wsDest.Cells(firstDataRow, colName), wsDest.Cells(lastDataRow, colName)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SortFields.Add Key:=wsDest.Range(wsDest.Cells(firstDataRow, colShiftSequence), wsDest.Cells(lastDataRow, colShiftSequence)), _
            SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
        .SetRange wsDest.Range(wsDest.Cells(firstDataRow, 1), wsDest.Cells(lastDataRow, colShiftSequence))
        .Header = xlNo
        .MatchCase = False
        .Orientation = xlTopToBottom
        .Apply
    End With
End Sub

Private Function CollectionContains(col As Collection, ByVal value As String) As Boolean
    Dim item As Variant
    For Each item In col
        If item = value Then
            CollectionContains = True
            Exit Function
        End If
    Next item
    CollectionContains = False
End Function

Private Function GetManualInvoiceDateValue() As Variant
    Dim cleanedValue As String
    Dim dateParts() As String

    cleanedValue = Replace(MANUAL_INVOICE_DATE, "{", "")
    cleanedValue = Replace(cleanedValue, "}", "")
    cleanedValue = Trim$(cleanedValue)

    dateParts = Split(cleanedValue, "/")

    If UBound(dateParts) = 2 Then
        If IsNumeric(dateParts(0)) And IsNumeric(dateParts(1)) And IsNumeric(dateParts(2)) Then
            GetManualInvoiceDateValue = DateSerial(CLng(dateParts(2)), CLng(dateParts(1)), CLng(dateParts(0)))
            Exit Function
        End If
    End If

    If IsDate(cleanedValue) Then
        GetManualInvoiceDateValue = DateValue(cleanedValue)
    Else
        GetManualInvoiceDateValue = MANUAL_INVOICE_DATE
    End If
End Function

Private Function GetSortedCollectionValues(ByVal col As Collection) As Variant
    Dim values() As String
    Dim item As Variant
    Dim i As Long
    Dim j As Long
    Dim temp As String

    ReDim values(1 To col.Count)

    i = 1
    For Each item In col
        values(i) = CStr(item)
        i = i + 1
    Next item

    For i = LBound(values) To UBound(values) - 1
        For j = i + 1 To UBound(values)
            If StrComp(values(i), values(j), vbTextCompare) > 0 Then
                temp = values(i)
                values(i) = values(j)
                values(j) = temp
            End If
        Next j
    Next i

    GetSortedCollectionValues = values
End Function

Private Sub ReorderSheetsAlphabetically(ByVal sortedClients As Variant)
    Dim i As Long
    Dim wsTarget As Worksheet

    On Error Resume Next
    Set wsTarget = ThisWorkbook.Worksheets(BuildClientSheetName(CStr(sortedClients(LBound(sortedClients)))))
    On Error GoTo 0

    If wsTarget Is Nothing Then Exit Sub
    wsTarget.Move After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count)

    For i = LBound(sortedClients) + 1 To UBound(sortedClients)
        On Error Resume Next
        Set wsTarget = ThisWorkbook.Worksheets(BuildClientSheetName(CStr(sortedClients(i))))
        On Error GoTo 0

        If Not wsTarget Is Nothing Then
            wsTarget.Move After:=ThisWorkbook.Worksheets(BuildClientSheetName(CStr(sortedClients(i - 1))))
        End If

        Set wsTarget = Nothing
    Next i
End Sub

Private Sub InsertCompanyLogo(ByVal ws As Worksheet)
    Const LOGO_SHAPE_NAME As String = "DorvictCompanyLogo"

    Dim targetRange As Range
    Dim logoShape As Shape
    Dim targetLeft As Double
    Dim targetTop As Double
    Dim targetWidth As Double
    Dim targetHeight As Double

    If Dir$(COMPANY_LOGO_PATH) = "" Then Exit Sub

    On Error Resume Next
    ws.Shapes(LOGO_SHAPE_NAME).Delete
    On Error GoTo 0

    Set targetRange = ws.Range("C1:D4")
    targetLeft = targetRange.Left
    targetTop = targetRange.Top
    targetWidth = targetRange.Width
    targetHeight = targetRange.Height

    Set logoShape = ws.Shapes.AddPicture( _
        Filename:=COMPANY_LOGO_PATH, _
        LinkToFile:=msoFalse, _
        SaveWithDocument:=msoTrue, _
        Left:=targetLeft, _
        Top:=targetTop, _
        Width:=-1, _
        Height:=-1)

    With logoShape
        .Name = LOGO_SHAPE_NAME
        .LockAspectRatio = msoTrue

        If .Width > targetWidth Then .Width = targetWidth
        If .Height > targetHeight Then .Height = targetHeight

        .Left = targetLeft + (targetWidth - .Width) / 2
        .Top = targetTop + (targetHeight - .Height) / 2
        .Placement = xlMoveAndSize
    End With
End Sub
